# Poland conquered... honor MR Pact?
country_event = {
	id = austria.12
	title = austria.12.t
	desc = austria.12.d
	picture = GFX_report_event_german_troops
	
	fire_only_once = yes

	trigger = {
		tag = GER
		has_global_flag = aus_yes_pact
		AUS = { exists = yes }
		NOT =  { has_war_with = AUS }
		OR = {
			OR = {
				owns_state = 89
				owns_state = 88
			}
			AND = {
				POL = { has_capitulated = yes }
				OR = {
					controls_state = 88
					controls_state = 89
				}
			}
			OR = {
				any_other_country = {
					is_in_faction_with = GER
					GER = {
						is_faction_leader = yes
					}
					OR = {
						owns_state = 89
						owns_state = 88
						AND = {
							POL = { has_capitulated = yes }
							OR = {
								controls_state = 89
								controls_state = 88
							}
						}		
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		ai_chance = {
			base = 1
		}
		name = austria.12.a
		if = {
			limit = {
				88 = {
					OR = {
						is_controlled_by = GER
						any_other_country = {
							controls_state = 88
							is_in_faction_with = GER
							GER = { is_faction_leader = yes }
						}
					}
				}
			}
			AUS = { transfer_state = 88 }
		}
		if = {
			limit = {
				89 = {
					OR = {
						is_controlled_by = GER
						any_other_country = {
							controls_state = 89
							is_in_faction_with = GER
							GER = { is_faction_leader = yes }
						}
					}
				}
			}
			AUS = { transfer_state = 89 }
		}
		AUS = { country_event = austria.13 }
	}
	option = {
		name = austria.12.b
		ai_chance = {
			base = 1
			modifier = {
				add = -1
				is_historical_focus_on = yes
			}
		}
		AUS = { country_event = germany.16 }
		clr_global_flag = sov_yes_pact
		diplomatic_relation = {
			country = AUS
			relation = non_aggression_pact
			active = no
		}
		AUS = {
			add_state_claim = 88
			add_state_claim = 89
		}
	}
}
